SELECT * FROM PRACOWNIK WHERE PENSJA>1300;

SELECT K.IMIE,K.NAZWISKO FROM KLIENT K
JOIN USLUGA_FRYZJERSKA UF ON UF.ID_KLIENT=K.ID_KLIENT
JOIN LOKAL L ON L.ID_LOKALU=UF.ID_LOKALU
WHERE L.NAZWA='Trwa³a';

SELECT K.IMIE,K.NAZWISKO,RU.NAZWA FROM KLIENT K
JOIN USLUGA_FRYZJERSKA UF ON UF.ID_KLIENT=K.ID_KLIENT
JOIN RODZAJ_USLUGI RU ON UF.ID_RODZAJU=RU.ID_RODZAJU
GROUP BY  K.IMIE,K.NAZWISKO,RU.NAZWA;
